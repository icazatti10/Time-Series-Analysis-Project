def find_best_arima(series, p_max, d, q_max):

    max_ar=list(range(0,p_max))
    max_ma=list(range(0,q_max))
    
    aicc_matrix = np.full((len(max_ar), len(max_ma)), np.inf)
    bic_matrix  = np.full((len(max_ar), len(max_ma)), np.inf)
    
    n = len(series)
    
    for p in max_ar:
        for q in max_ma:
            try:
                with warnings.catch_warnings():
                    warnings.simplefilter("ignore")
                    fit_result = ARIMA(series, order=(p, d, q)).fit()
    
                k = len(fit_result.params)
                aicc = fit_result.aic + (2 * k**2 + 2 * k) / (n - k - 1)
                bic = fit_result.bic
    
                aicc_matrix[p, q] = aicc
                bic_matrix[p, q]  = bic
                
            except Exception:
                # Skip failed fits silently
                pass

    p_best_aicc, q_best_aicc = np.unravel_index(np.argmin(aicc_matrix), aicc_matrix.shape)
    best_aicc = aicc_matrix[p_best_aicc, q_best_aicc]

    p_best_bic, q_best_bic = np.unravel_index(np.argmin(bic_matrix), bic_matrix.shape)
    best_bic = bic_matrix[p_best_bic, q_best_bic]
    
    print("Best by AICc:", (p_best_aicc, d, q_best_aicc), "AICc:", best_aicc)
    print("Best by BIC :", (p_best_bic,  d, q_best_bic),  "BIC :", best_bic)

def find_best_garch(residuals, p_max=3, q_max=3, dist="t", mean="Zero"):
    final_bic = np.inf
    final_garch_order = None
    best_fit = None

    residuals = residuals.dropna()

    for p in range(p_max):
        for q in range(q_max):

            # GARCH(0,0) is usually not valid/useful
            if p == 0 and q == 0:
                continue

            try:
                with warnings.catch_warnings():
                    warnings.simplefilter("ignore")

                    garch_model = arch_model(
                        residuals,
                        vol="GARCH",
                        p=p,
                        q=q,
                        dist=dist,
                        mean=mean
                    )

                    garch_fit = garch_model.fit(disp="off")

                current_bic = garch_fit.bic

                if np.isfinite(current_bic) and current_bic < final_bic:
                    final_bic = current_bic
                    final_garch_order = (p, q)
                    best_fit = garch_fit

            except Exception:
                continue

    if final_garch_order is not None:
        print(
            f"Best GARCH order: {final_garch_order} "
            f"with BIC = {final_bic:.6f}"
        )
    else:
        print("No valid GARCH fit found on the given grid.")

def find_best_arma_for_fixed_garch(series, p_max=6, q_max=6, garch_p=1, garch_q=1, d=0, dist="t",scale=100.0):
    orders = []

    series = series.dropna().astype(float)

    for p in range(p_max):
        for q in range(q_max):
            try:
                with warnings.catch_warnings():
                    warnings.simplefilter("ignore")

                    arma_fit = ARIMA(series, order=(p, d, q)).fit()
                    resid_scaled = arma_fit.resid.dropna() * scale

                    garch_model = arch_model(resid_scaled, mean="Zero", vol="GARCH", p=garch_p, q=garch_q, dist=dist, rescale=False)

                    garch_fit = garch_model.fit(disp="off")

                bic_val = garch_fit.bic
                aic_val = garch_fit.aic

                if np.isfinite(bic_val):
                    orders.append({
                        "p": p,
                        "d": d,
                        "q": q,
                        "garch_p": garch_p,
                        "garch_q": garch_q,
                        "arma_aic": arma_fit.aic,
                        "arma_bic": arma_fit.bic,
                        "garch_aic": aic_val,
                        "garch_bic": bic_val
                    })

            except Exception:
                continue

    orders_df = pd.DataFrame(orders)

    if orders_df.empty:
        print("No valid ARMA-GARCH fits found.")
        return None, None

    orders_df = orders_df.sort_values("garch_bic").reset_index(drop=True)

    best_p = int(orders_df.loc[0, "p"])
    best_q = int(orders_df.loc[0, "q"])

    print(f"Best ARMA by fixed GARCH({garch_p},{garch_q}) -  ARMA({best_p},{best_q})")
    print(orders_df.head(5))

def rolling_arma_garch_1step(train_series, test_series, p, q, gp, gq, dist="t", scale=100.0):
    # assume train_series and test_series are pandas Series with indexes
    idx = test_series.index
    n = len(test_series)
    mean_out = np.empty(n, dtype=float)
    sigm_out = np.empty(n, dtype=float)

    for f in range(n):
        y = pd.concat([train_series, test_series.iloc[:f]]).astype(float)

        with suppress_failed_hessian():
            arma_fit = ARIMA(y, order=(p, 0, q)).fit()

        resid_scaled = arma_fit.resid.dropna() * scale

        with suppress_failed_hessian():
            gfit = arch_model(resid_scaled, mean="Zero", vol="GARCH", p=gp, q=gq, dist=dist, rescale=False).fit(disp="off")

        mu_fc = arma_fit.get_forecast(steps=1).predicted_mean.iloc[-1]
        var_fc = gfit.forecast(horizon=1).variance.values[-1, 0]
        sigma = np.sqrt(var_fc) / scale

        mean_out[f] = float(mu_fc)
        sigm_out[f] = float(sigma)

    mean_out[~np.isfinite(mean_out)] = 0.0
    return pd.Series(mean_out, index=idx), pd.Series(sigm_out, index=idx)

def evaluate_performance(true, pred, model_name="Model"):
    mspe = mean_squared_error(true, pred)
    mae = mean_absolute_error(true, pred)
    mape = np.mean(np.abs((true - pred) / (true + 1e-6)))
    pm = (np.sum((true+1e-6 - pred) ** 2) / np.sum((true+1e-6 - np.mean(true)) ** 2))

    print(f"=== {model_name} Performance ===")
    print(f"MSPE: {mspe:.4f}")
    print(f"MAE:  {mae:.4f}")
    print(f"MAPE: {mape:.4f}")
    print(f"PM:   {pm:.4f}")
    print("-" * 40)
